#include <iostream>
#include <cstdlib>
using namespace std;

int* randArray(int n) {
    int* arr = new int[n];

    for (int i = 0; i < n; i++) {
        arr[i] = rand() % 100;
    }

    return arr;
}

int main() {
    int* arr = randArray(100);

    for (int i = 1; i < 100; i += 2) {
        cout << arr[i] << " ";
    }
    cout << endl;

    for (int i = 0; i < 100; i++) {
        if (i % 3 == 0 && i % 5 == 0) {
            cout << arr[i] << " ";
        }
    }
    cout << endl;

    for (int i = 0; i < 100; i++) {
        cout << arr[i] << endl;
    }

    delete[] arr;

    return 0;
}
